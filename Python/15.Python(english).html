Now let is see Python os Module. it is a inbuilt module by which you can perform system operations such as opening a app, deleting a file, renaming, etc. 
import os
print(os.getcwd())  # this will print cwd -> current working directory
os.chdir('/path/to/new/directory') # this will change your current working directory
print(os.listdir('.'))  # list all files in the directory  '.' represents the current directory 
os.mkdir('new_folder')   # make directory creates a new directoryand new-folder represents name of the new directory/folder
os.remove('folder/file name')  # this removes the files
os.rename('old name' , 'new name')  # renames the file
os.system("command")  # runs commands in the system powershell

let us write some program keeping this directory in mind

we have this as own initial directory



import os

# Print the current working directory
print("Current working directory:", os.getcwd())

# Print the list of files in the current directory
print("Initial directory contents:", os.listdir('.'))

# Create new files
print("\n************* Creating new files *********")
for i in range(5):
    file_name = f"cute_file_{i}"
    try:
        with open(file_name, 'x') as file:  # 'x' mode for exclusive creation
            file.write(f"Hello, world! This is file {file_name}")
        print(f"File created: {file_name}")
    except FileExistsError:
        print(f"Error: The file '{file_name}' already exists.")

# Identify and print 'cute' files
print("\n************* Identifying 'cute' files *********")
for fileName in os.listdir('.'):
    if "cute" in fileName:
        print(f"Identified a 'cute' file: {fileName}")

# Identify and delete unwanted files
print("\n************* Deleting 'unwanted' files *********")
for fileName in os.listdir('.'):
    if "unw" in fileName.lower():
        print(f"Deleting unwanted file: {fileName}")
        os.remove(fileName)

# Rename the remaining files in a structured way
print("\n************* Renaming remaining files *********")
for index, fileName in enumerate(os.listdir('.')):
    if fileName != "configuration.py" and 'cute' not in fileName:  # Ensure not to rename the running script
        new_name = f"important_file_{index}"
        os.rename(fileName, new_name)
        print(f"Renamed '{fileName}' to '{new_name}'")

# Print the final list of files in the directory
print("\nFinal directory contents:", os.listdir('.'))
