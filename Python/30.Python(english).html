CREATE TABLE Production(
ID INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
name TEXT    -- we will not specify the length it can be anything

-- primary key makes the ID primary key
-- AUTOINCREMENT tells that we do not have to specify it. you do it yourself generate it yourself
-- UNIQUE tells that ID should be UNIQUE

-- lets make it complex

SELECT Production.name , Genre.name , Artist.name , Movie.*
FROM Production JOIN Genre JOIN Artist JOIN Movie
ON Production.ID = Artist.Production_id and Genre.ID = Movie.Genre_id and Artist.ID = Movie.Artist_id;
