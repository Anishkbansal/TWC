        return 1    # because zero factorial is one. and if by chance n-1 == 0 then we dont multiply the factorial by zero. or even go the negative integers.

    for i in range(1, n + 1):
        result = result * i
    return result

def fibonacci(n):
    # Base cases
    if n <= 0:
        return 0
    elif n == 1:
        return 1
    
    # Recursive case
    return fibonacci(n - 1) + fibonacci(n - 2)

n = int(input("what number of the series you want to see? "))
print(fibonacci(n))
