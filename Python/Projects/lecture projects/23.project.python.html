import speech_recognition as sr
import pyttsx3

def listen():
    recognizer = sr.Recognizer()        # initialising the speech recognizer
    with sr.Microphone() as source:     # telling to use our mic as a source for the speech
        print("Listening...")
        audio_data = recognizer.listen(source)         # listen the speech
        print("Recognizing...")
        try:          # trying to understand and convert the speech
            text = recognizer.recognize_google(audio_data) 
            return text 
        except sr.UnknownValueError:
            return "Sorry, I did not understand that."
        except sr.RequestError:
            return "Sorry, my speech service is down."

def processing(text):
    processed_text = f"Did you say {text}?"    # processing text. we will use ai to process later
    print(processed_text)
    return processed_text

def speak(text):
    speaker = pyttsx3.init()     # initialising text to speach
    speaker.say(text)       # telling it to say text
    speaker.runAndWait()    # i asked ai abut it and it responded ->, Without engine.runAndWait(), the program might finish executing before the speech engine has had a chance to speak the text. This method ensures that the speech is actually output before the program continues or finishes execution.


if __name__ == "__main__":
    text = listen()      # calling listen function which convert speech to text and return that text.
    print(f"Recognized Text: {text}")   # printing the text

    response = processing(text)     # processing the text 
    print(response)
    speak(response)       # printng the text
