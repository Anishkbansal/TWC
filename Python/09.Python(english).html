MyList = ["hello", 25, "Peepo", None, True, [56, 10] ]   # we enclose list inside square brackets i.e []  . we seperate the elements by commas
# the number of element is called index. in lists, the starting index is zero (0) which means that the zeroeth element (whoose index is zero) in this list is "hello", and its fifth element is [56, 10] which is a list itself. 

print(MyList)
print(MyList[2])     # Name[x] this is the syntax of getting the element at the given index. if you replace Name with the name of your list and x with the number of index you want to get.
print(MyList[5][0]) # this code here means that get the element at index 5, and give zeroeth index element of that element
print(MyList[0][2])  # String also behaves as a list in the backend (list of characters ) so this code will give 2nd index of element at the first index of MyList
print(MyList[2][-2])  # you can even put negative indexes in this. 0 -1 will send to end of the list. so -1 index will be the last element of the list. -2 -> 2nd last. i will put up and image of this. 
print(MyList[0:2])   # Now this is called list slicing. we slice out a part from the list. and this is how we tell Python about what part to slice. 
# Name[initial:ending:iterate ] replacing Name with name of your list, .
# initial with index you want to start slicing from, 
# ending with end you want to stop at + 1  (+1 because it works like a for loop and check index less than the index provided. example if index 3 is at ending place, it will print till index 2. )
# replace iterate with the number you want to jump through indexes like in a for loop. its default value is 1


print(MyList[:1], MyList[3:])  # if not initial index is provided, it will start from the list's staring. and if no ending is told, it will print till end of the list
print(MyList[2::2])   # the ending not provided so it will print till end of list and iterate is provided which is 2. so Python will print every other element.

Now lets move on to List Methods.List is Mutable. Mutable means we can change its value.
            


Lst = [1,35,565,65,45,56,34,21]
print(Lst)
Lst.append(103)   # append(x) add an item to the list at the end where x is the item you want to append the list with 
Lst.append(103+3)
Lst.append(10>30)
print("List after using 3 append", Lst)
Lst.extend([56,56,True]) # extend(l) add an List of items to the list. where l is the list. extend() takes iterable things meaning whose value can be returned one by one using a for loop like list,strings,tuples etc
# if you are confused in iterable, try to print each number of a big number one by one like 1452 becomes 1     2    5    2   etc. and do the same with a list or string

print("List after using extend", Lst)

Lst.insert(2, "imposter")  # insert(x,y) is like big brother of append which also allowes you to specify the index you want to place the new element at
# here, x or first argument will be the index number and second will be the item you want to insert
print("list after insert", Lst)

if("imposter" in Lst):
    Lst.remove("imposter")   # remove(x)  where x is the element you want to remove. this function removes the first occurence of item in the list. If the item is not present in the list IT WILL THROW ERROR
# so to avoid error we usually wrap the remove(x) in an if else statement. like did here
print("list after remove", Lst)

item_popped = Lst.pop(4)   # works same as remove but more flexible. and also it TAKES INDEX NUMBER AS ARGUMENT NOT THE ELEMENT ITSELF LIKE remove()...... (1) if no argument is provided, removes the last item. (2) returns the item removed which means you can store it in a variable
print("popped ", item_popped, "from the list", Lst)

print("the number 65 occurs at index, ", Lst.index(65))  # returns the index of first occurenece of the item given as argument in the list. ERROR IF(ITEM NOT IN LIST) 

print("the number of times 56 occurs in this list is ", Lst.count(56))   # as the name suggest, takes item as argument and count no of times an item appears in the list

Lst.reverse()   # works as expected. flips the order of the list. meaning first item at last and last one at the first
print("List after reverse", Lst)

Lst.sort()  # sort the list form lowest to highest . You can also sort lists with strings in it. Try it please
print("list after sorting ", Lst)
Lst.sort(reverse = True)  # is the argument reverse=True is given it will sort the opposite. Highest to Lowest
print("list after reverse sorting, " Lst)

a = Lst    # FOCUS HERE... this will NOT create new list called a with same element as Lst. infact this will give another name to Lst.

b = Lst.copy()   # if you actually wantt to create a copy of a list you can use copy(). now b is a different list.

listLength = Lst.len()   # return number of items in the list as an integer
print("length of the list is ", listLength)

bo = Lst.contains("imposter2")    # contains() checks is the item specified is present in the list or not . it returns a BOOLEAN value.
