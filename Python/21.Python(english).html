import urllib.request
import json

response = urllib.request.urlopen('https://anishbansal.pythonanywhere.com/data/TWC.json')
json_data = ''
for line in response:
    json_data += line.decode()             # getting json data
print(json_data)

print("************************")         # added starts to beautify the output

dict = json.loads(json_data)              # in xml we had ET.fromstring(xml_data)     in json we have json.load()

print(dict , type(dict))       # now the data loaded is a dictionary

print("********************")

print('database Name: ' , dict["databaseName"])      # so we can run operations of dictionary on it
 
print("user1 username: " , dict["users"]["UserID-1"]["username"])

print("user 2 details: ", dict["users"]["UserID-2"])     # this outputs dictionary as the value of UserID-2 is a dictionary
